BASE = http://localhost:3000

login             = POST   http://localhost:3000/api/v4/user/login
googleLogin       = POST   http://localhost:3000/api/v4/user/google-login
getMe             = GET    http://localhost:3000/api/v4/user/me
getUserById       = GET    http://localhost:3000/api/v4/user/get-user/:id
updateProfile     = PUT    http://localhost:3000/api/v4/user/update-profile/:id
deleteUser        = DELETE http://localhost:3000/api/v4/user/delete-user/:id

updateSeats       = PATCH  http://localhost:3000/api/v4/rides/update-seats/:id
